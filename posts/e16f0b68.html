#01 Eloquent ORM 中 Relationships[1] 与 Serialization[2] 特性结合产生的坑

namespace App\Models;

use Illuminate\Database\Eloquent\Model;

class User extends Model
{
    protected $table = 'users';

    public $timestamps = true;

    public $dateFormat = 'U';

    protected $fillable = [
        //
    ];

    protected $hidden = [
        //
    ];

    protected $appends = [
        'room_id',
    ];

    public function room()
    {
        return $this->hasOne('App\Models\Room')->withDefault(['id'=>0])->whereLive(0);
    }

    public function getRoomIdAttribute()
    {
        if ($this->room) {
            return $this->room['id'];
        }

        return 0;
    }

坑点: appends的字段取自relationship表的字段,引起无论是否需要查询关联子表，只要使用该model的序列化方法toArray or toJson, 都会引起关联表查询.

解决方式: 尽量不要定义来自关联子表的字段到appends里.
