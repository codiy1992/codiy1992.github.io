快速搭建启动VPN服务

codiy 2019-03-06 10:47:58
02-Docker容器
docker, docker-compose
docker-compose.yml

---
version: '3'

services:
  vpn:
    image: hwdsl2/ipsec-vpn-server
    restart: always
    env_file:
      - ./.env
    environment:
      - VPN_IPSEC_PSK=wangle520
      - VPN_USER=wangle
      - VPN_PASSWORD=wangle12958
    ports:
      - "500:500/udp"
      - "4500:4500/udp"
    privileged: true

  shadowsocks:
    image: shadowsocks/shadowsocks-libev
    ports:
      - "4433:8388/tcp"
      - "4433:8388/udp"
    environment:
      - METHOD=aes-256-cfb
      - PASSWORD=6Et80G4N5N8Lh3kw
    restart: always
.env
